CREATE TABLE zoologico (
  IDZoo INT PRIMARY KEY,
  nombre VARCHAR(100) NOT NULL,
  ciudad VARCHAR(100) NOT NULL,
  pais VARCHAR(50) NOT NULL,
  tamagno FLOAT NOT NULL,
  presupuesto DOUBLE NOT NULL
);

CREATE TABLE especies (
  IDEsp INT PRIMARY KEY,
  nombreVulgar VARCHAR(100) NOT NULL,
  nombreScience VARCHAR(100) NOT NULL,
  familia VARCHAR(100) NOT NULL,
  dangerExtinct BIT NOT NULL
);

CREATE TABLE animales (
  IDAnimal INT PRIMARY KEY,
  zoo INT NOT NULL, esp INT NOT NULL,
  sex BIT NOT NULL, born DATE NOT NULL,
  pais VARCHAR(50) NOT NULL, continente VARCHAR(50) NOT NULL,
  FOREIGN KEY (zoo) REFERENCES zoologico(IDZoo),
  FOREIGN KEY (esp) REFERENCES especies(IDEsp)
);
